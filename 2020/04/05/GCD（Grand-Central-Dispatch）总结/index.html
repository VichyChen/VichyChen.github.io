GCD（Grand Central Dispatch）总结

1、常用

2、一次性执行

3、延迟两秒执行

4、执行某个代码片段N次

5、timer

6、自定义queue
// 创建串行队列
dispatch_queue_t queue = dispatch_queue_create(“thread”, DISPATCH_QUEUE_SERIAL);
// 创建并行队列
dispatch_queue_t queue = dispatch_queue_create(“thread”, DISPATCH_QUEUE_CONCURRENT);

7、让两个异步线程并行执行，等两个线程都结束，结果汇总
dispatch_group_t group = dispatch_group_create();
dispatch_queue_t queue = dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0);
dispatch_group_async(group, queue, ^{
sleep(2);
NSLog(@”group1”);
});
dispatch_group_async(group, queue, ^{
sleep(1);
NSLog(@”group2”);
});
dispatch_group_notify(group, dispatch_get_main_queue(), ^{
NSLog(@”update UI”);
});

// output
// 23:08:07.499398+0800 group2
// 23:08:08.497703+0800 group1
// 23:08:08.498081+0800 update UI

10、同步栅栏dispatch_barrier_sync、异步栅栏dispatch_barrier_async
dispatch_barrier_sync 需要等待栅栏执行完才会执行栅栏后面的任务。
dispatch_barrier_async 无需等待栅栏执行完，会继续往下走。

dispatch_queue_t queue = dispatch_queue_create("thread", DISPATCH_QUEUE_CONCURRENT);
dispatch_async(queue, ^{
    sleep(1);
    NSLog(@"任务1完成");
});
dispatch_async(queue, ^{
    sleep(3);
    NSLog(@"任务2完成");
});
dispatch_barrier_sync(queue, ^{
    NSLog(@"执行barrier");
});
dispatch_async(queue, ^{
    NSLog(@"任务3完成");
});
NSLog(@"dispatch_barrier_sync阻塞线程");

// output
// 12:14:41.147217+0800 任务1完成
// 12:14:43.145697+0800 任务2完成
// 12:14:43.145914+0800 执行barrier
// 12:14:43.146059+0800 dispatch_barrier_sync阻塞线程
// 12:14:43.146067+0800 任务3完成

dispatch_queue_t queue = dispatch_queue_create("thread", DISPATCH_QUEUE_CONCURRENT);
dispatch_async(queue, ^{
    sleep(1);
    NSLog(@"任务1完成");
});
dispatch_async(queue, ^{
    sleep(3);
    NSLog(@"任务2完成");
});
dispatch_barrier_async(queue, ^{
    NSLog(@"执行barrier");
});
dispatch_async(queue, ^{
    NSLog(@"任务3完成");
});
NSLog(@"dispatch_barrier_async不阻塞线程");

// output
// 12:01:29.514468+0800 dispatch_barrier_async不阻塞线程
// 12:01:30.518375+0800 任务1完成
// 12:01:32.519471+0800 任务2完成
// 12:01:32.519861+0800 执行barrier
// 12:01:32.520156+0800 任务3完成

同时执行，结果汇总

dispatch_group_enter、dispatch_group_leave
dispatch_group_t group = dispatch_group_create();
dispatch_group_enter(group);
[self.request1 startWithComplete:^ {
// do something
dispatch_group_leave(serviceGroup);
}];
dispatch_group_enter(group);
[self.request2 startWithComplete:^ {
// do something
dispatch_group_leave(serviceGroup);
}];
dispatch_group_notify(group, dispatch_get_main_queue(),^{
NSLog(@”update UI”);
});

信号量
dispatch_group_t group = dispatch_group_create();
dispatch_group_async(group, dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
dispatch_semaphore_t sema = dispatch_semaphore_create(0);
[self.request1 startWithComplete:^{
dispatch_semaphore_signal(sema);
}];
dispatch_semaphore_wait(sema, DISPATCH_TIME_FOREVER);
});
dispatch_group_async(group, dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{

循环网络请求 同时进行 统一回调
信号量
dispatch_group_t group = dispatch_group_create();
for (int i = 0 ; i < 5; i++) {
dispatch_group_async(group, dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
dispatch_semaphore_t sema = dispatch_semaphore_create(0);
// 模拟请求 ↓
dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
sleep(3);
NSLog(@”任务%d完成”,i);
dispatch_semaphore_signal(sema);
});
// 模拟请求 上
dispatch_semaphore_wait(sema, DISPATCH_TIME_FOREVER);
});
}
dispatch_group_notify(group, dispatch_get_main_queue(), ^{
NSLog(@”全部搞完了”);
});

dispatch_group_enter、dispatch_group_leave
dispatch_group_t group = dispatch_group_create();
for (int i = 0 ; i < 5; i++) {
dispatch_group_enter(group);
// 模拟请求 ↓
dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
sleep(3);
NSLog(@”任务%d完成”,i);
dispatch_group_leave(group);
});
// 模拟请求 ↑
}
dispatch_group_notify(group, dispatch_get_main_queue(), ^{
NSLog(@”全部搞完了”);
});

dispatch_semaphore_t sema = dispatch_semaphore_create(0);
dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
for (int i = 0 ; i < 5; i++) {
NSLog(@”开始%d”,i);

sleep(3);
NSLog(@”任务%d完成”,i);
dispatch_semaphore_signal(sema);
});
// 模拟请求 上
dispatch_semaphore_wait(sema, DISPATCH_TIME_FOREVER);
}

   NSLog(@"全部搞完了");
});

dispatch_group_enter、dispatch_group_leave
dispatch_group_t group = dispatch_group_create();
for (int i = 0 ; i < 5; i++) {
dispatch_group_enter(group);
// 模拟请求 ↓
dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
sleep(6 - i);
NSLog(@”任务%d完成”,i);
dispatch_group_leave(group);
});
// 模拟请求 ↑
dispatch_group_wait(group, DISPATCH_TIME_FOREVER); // 顺序执行与同步执行的不同点
}
dispatch_group_notify(group, dispatch_get_main_queue(), ^{
NSLog(@”全部搞完了”);
});
